  <action>
    Install Prisma and define the domain schema for the entire app.

    Step 1 — Install dependencies:
    ```
    npm install prisma @prisma/client
    npx prisma init --datasource-provider sqlite
    ```

    This creates prisma/schema.prisma and .env with DATABASE_URL="file:./dev.db".

    Step 2 — Write prisma/schema.prisma with this exact schema:

    ```prisma
    generator client {
      provider = "prisma-client-js"
    }

    datasource db {
      provider = "sqlite"
      url      = env("DATABASE_URL")
    }

    model Campaign {
      id                  String       @id @default(cuid())
      name                String
      planningWindowStart DateTime?
      planningWindowEnd   DateTime?
      createdAt           DateTime     @default(now())
      updatedAt           DateTime     @updatedAt
      playerSlots         PlayerSlot[]
    }

    model PlayerSlot {
      id                    String              @id @default(cuid())
      campaignId            String
      name                  String
      inviteToken           String              @unique @default(cuid())
      createdAt             DateTime            @default(now())
      updatedAt             DateTime            @updatedAt
      campaign              Campaign            @relation(fields: [campaignId], references: [id], onDelete: Cascade)
      availabilityEntries   AvailabilityEntry[]
    }

    // type: "weekly" = recurring day-of-week pattern
    // type: "override" = specific date exception
    model AvailabilityEntry {
      id           String     @id @default(cuid())
      playerSlotId String
      type         String     // "weekly" | "override"
      dayOfWeek    Int?       // 0=Sun, 1=Mon, ..., 6=Sat (used when type="weekly")
      date         DateTime?  // specific date (used when type="override")
      timeOfDay    String?    // "morning" | "afternoon" | "evening"
      status       String     // "free" | "busy"
      createdAt    DateTime   @default(now())
      updatedAt    DateTime   @updatedAt
      playerSlot   PlayerSlot @relation(fields: [playerSlotId], references: [id], onDelete: Cascade)
    }
    ```

    Step 3 — Push schema to create the SQLite database:
    ```
    npx prisma db push
    ```

    Step 4 — Generate the Prisma client:
    ```
    npx prisma generate
    ```

    Step 5 — Create src/lib/prisma.ts (singleton client for Next.js dev hot-reload safety):
    ```typescript
    import { PrismaClient } from '@prisma/client'

    const globalForPrisma = globalThis as unknown as {
      prisma: PrismaClient | undefined
    }

    export const prisma =
      globalForPrisma.prisma ??
      new PrismaClient({
        log: ['query'],
      })

    if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma
    ```

    Step 6 — Create .env.example (safe to commit, no secrets):
    ```
    DATABASE_URL="file:./dev.db"
    ```

    Add to .gitignore (if not already present):
    ```

    prisma/dev.db
    prisma/dev.db-journal
    .env
    ```
